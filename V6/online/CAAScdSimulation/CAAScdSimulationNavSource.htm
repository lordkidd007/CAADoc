'-------------- Sample MACRO written in Basic Script ------------------------------------------ 
'This sample retrieves the Root Feature of Scenario or Result Representations and the simulated Model 
'Before starting this macro, opens a simulation structure containing a Scenario or Result rep. 
'Steps:
'   1- Retrieves the current Editor
'   2- Retrieves the Simulation Reference from the Editor
'   3- Retrieves the Simulated Model 
'   4- Retrieving the Specifications Collection  
'   5- Retrieving the Scenario Representation under the Scenarios Category 
'   6- Retrieving the Root Feature of the Rep Scenario
'   7- Retrieving the Results Collection  
'   8- Retrieving the Result Representation under the Results Category 
'   9- Retrieving the Root Feature of the Rep Result



'''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''''

Sub CATMain()


'   1- Gets the current Editor 
        Dim oSimulationEditor As  Editor
        Set oSimulationEditor = CATIA.ActiveEditor

'   2- Gets the Simulation Reference from the Editor 
        Dim oSimulationRoot As  SimulationReference
        Set oSimulationRoot = oSimulationEditor.ActiveObject


'   3- Retrieves the Simulated Model 
        Dim oPLMModel  As  AnyObject
        Set oPLMModel = oSimulationRoot.Model

'   4- Retrieving the Specifications Collection 
       Dim oSpecifications  As  SimulationSpecifications
       Set oSpecifications = oSimulationRoot.Specifications


'   5- Retrieving the Scenario Representation under the Scenarios Category
       Dim oSimulationRepSpec  As  SimulationRepresentation
       Set oSimulationRepSpec = oSpecifications.Item(1)
 
'   6- Retrieving the Root Feature of the Rep
       'Available when the root feature will implement CATIABase...
       'Dim oRoot As AnyObject
       'Set oRoot = oSimulationRepSpec .Root
 
'   7- Retrieving the Results Collection 
       Dim oResults   As  SimulationResults
       Set oResults  = oSimulationRoot.Results


'   8- Retrieving the Result Representation under the Results Category
       Dim oSimulationRepRes  As  SimulationRepresentation
       Set oSimulationRepRes = oResults.Item(1)
 
'   9- Retrieving the Root Feature of the Rep
       'Available when the root feature will implement CATIABase...
       'Dim oRoot As AnyObject
       'Set oRoot = oSimulationRepRes .Root


        
End Sub
