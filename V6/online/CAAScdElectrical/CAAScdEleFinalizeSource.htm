Option Explicit

' ***********************************************************************
'   Purpose:      Post process the imported data using Finalize on interface CATIAElecImportFinalizerService
'   Assumtions:   None  
'   Languages:    VBSCRIPT
'   Locales:      English 
'   CATIA Level:  V6R2010x
' ***********************************************************************

Sub CATMain()

' --------------------------
' Retrieve the variable MyElecImportFinalizerService of type ElecImportFinalizer from the active object(Root).
' --------------------------

Dim oElecService As ElecImportFinalizerService
Set oElecService = CATIA.ActiveEditor.GetService("ElecImportFinalizer")

' --------------------------
' Call Finalize on the variable MyElecImportFinalizerService.
' --------------------------

oElecService.Finalize

End Sub
